networks=CREATE TABLE APP.NETWORKS (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), NETWORK VARCHAR(25) NOT NULL, URL VARCHAR(256) NOT NULL)
webjauth=CREATE TABLE APP.WEBJAUTH (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), MYAUTH VARCHAR(256) NOT NULL, SECRET VARCHAR(256) NOT NULL)
accounts=CREATE TABLE APP.ACCOUNTS (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), NETWORK VARCHAR(25) NOT NULL, PUBLIC_KEY BLOB NOT NULL, PRIVATE_KEY BLOB NOT NULL)
